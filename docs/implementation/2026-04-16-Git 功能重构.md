## 一、背景与目标

### 1.1 背景

当前 SolarWire 编辑器的 Git 功能存在以下问题：

1. **VersionView 定位模糊**：未集成到 Tab 系统，用户无法通过 Tab 切换
2. **GitView 功能残缺**：仅支持基本的暂存/提交，缺少分支管理、Diff 查看等主流功能
3. **状态管理不一致**：部分视图使用 `selectionStore`，VersionView 使用内部状态
4. **无法精确显示版本历史**：无法区分 Markdown 文件中不同 SolarWire 代码块的修改记录

### 1.2 核心目标

构建一个**智能的、精确的、高性能的**版本历史查看系统，支持：

1. ✅ **精确过滤**：只显示与当前编辑内容相关的 Git 提交
2. ✅ **异步处理**：使用 Web Worker 后台分析，不阻塞 UI
3. ✅ **进度可视化**：状态栏 + VersionView 内部双进度条显示
4. ✅ **智能缓存**：避免重复分析，提升性能
5. ✅ **错误友好**：清晰的错误提示和操作建议

### 1.3 设计原则

- **不修改现有解析器和渲染器**：使用通用 Markdown 解析器（如 `marked`）
- **基于 AST 对比**：精确识别代码块是否被修改，而非简单的文本匹配
- **用户友好**：清晰的错误提示和操作建议
- **性能优先**：限制最大分析数量（50 个提交），使用缓存

```tsx
          <CommitItem 
            key={commit.hash} 
            commit={commit}
            onSelect={handleCommitSelect}

  );
```

### 4.2 useVersionHistory Hook

**文件位置**: `editor/src/shared/hooks/useVersionHistory.ts`

**核心职责**:
- 管理版本历史分析的状态
- 与 Web Worker 通信
- 缓存管理
- 进度更新

**实现代码**:
```typescript
interface UseVersionHistoryResult {
  analysisProgress: AnalysisProgress | null;
  matchingCommits: GitCommit[];
  error: string | null;
  suggestion: string | null;
  cancelAnalysis: () => void;
  isLoading: boolean;
}

export function useVersionHistory(
  filePath: string,
  snippet?: { code: string; snippetIndex: number }
): UseVersionHistoryResult {
  const [progress, setProgress] = useState<AnalysisProgress | null>(null);
  const [commits, setCommits] = useState<GitCommit[]>([]);
  const [error, setError] = useState<string | null>(null);
  const [suggestion, setSuggestion] = useState<string | null>(null);
  const [worker, setWorker] = useState<Worker | null>(null);
  
  const { isInitialized, getGitLog } = useGitStore();
  const versionCache = useVersionCache();
  
  useEffect(() => {
    // 1. 验证 !title 是否存在
    const titleValidation = validateTitle(filePath, snippet);
    if (!titleValidation.valid) {
      setError(titleValidation.error);
      setSuggestion(titleValidation.suggestion);
      return;
    }
    
    // 2. 检查缓存
    const cacheKey = buildCacheKey(filePath, snippet, titleValidation.title);
    const cached = versionCache.get(cacheKey);
    if (cached) {
      setCommits(cached.commits);
      return;
    }
    
    // 3. 获取 Git commits
    const gitCommits = await getGitLog(filePath);
    const limitedCommits = gitCommits.slice(0, 50); // 限制最多 50 个
    
    // 4. 创建 Worker
    const analyzerWorker = new Worker(
      new URL('../workers/git-diff-analyzer.worker.ts', import.meta.url)
    );
    setWorker(analyzerWorker);
    
    // 5. 监听 Worker 消息
    analyzerWorker.onmessage = (event) => {
      const { type, data } = event.data;
      
      if (type === 'progress') {
        setProgress(data);
        // 更新状态栏
        updateStatusBar(data);
      } else if (type === 'complete') {
        setProgress({ ...data, status: 'completed' });
        setCommits(data.matchingCommits);
        
        // 缓存结果
        versionCache.set(cacheKey, {
          commits: data.matchingCommits,
          timestamp: Date.now(),
          analyzedCount: data.analyzedCount,
          context: { filePath, snippet, title: titleValidation.title }
        });
      } else if (type === 'error') {
        setError(data.message);
        setSuggestion(data.suggestion);
      }
    };
    
    // 6. 发送分析任务
    analyzerWorker.postMessage({
      type: 'analyze',
      commits: limitedCommits,
      filePath,
      context: {
        type: snippet ? 'markdown-snippet' : 'solarwire-file',
        title: titleValidation.title,
        snippetIndex: snippet?.snippetIndex
      }
    });
    
    // 7. 清理
    return () => {
      analyzerWorker.terminate();
    };
  }, [filePath, snippet]);
  
  const cancelAnalysis = useCallback(() => {
    if (worker) {
      worker.postMessage({ type: 'cancel' });
      worker.terminate();
      setProgress(prev => prev ? { ...prev, status: 'cancelled' } : null);
    }
  }, [worker]);
  
  return {
    analysisProgress: progress,
    matchingCommits: commits,
    error,
    suggestion,
    cancelAnalysis,
    isLoading: progress?.status === 'running'
  };
}
```

### 4.3 GitDiffAnalyzer Web Worker

**文件位置**: `editor/src/workers/git-diff-analyzer.worker.ts`

**核心职责**:
- 后台分析 Git 提交
- 使用 marked 解析 Markdown
- 提取 SolarWire 代码块
- AST 对比识别代码块变化

**实现代码**:
```typescript
// git-diff-analyzer.worker.ts
import { marked } from 'marked';

interface WorkerMessage {
  type: 'analyze' | 'cancel';
  commits: GitCommit[];
  filePath: string;
  context: {
    type: 'solarwire-file' | 'markdown-snippet';
    title: string;
    snippetIndex?: number;
  };
}

interface WorkerResponse {
  type: 'progress' | 'complete' | 'error';
  data: any;
}

// 监听主线程消息
self.onmessage = async (event: MessageEvent<WorkerMessage>) => {
  const { type, commits, filePath, context } = event.data;
  
  if (type === 'cancel') {
    // 取消分析
    return;
  }
  
  try {
    const matchingCommits: GitCommit[] = [];
    let analyzedCount = 0;
    
    // 分析每个 commit
    for (let i = 0; i < commits.length; i++) {
      const commit = commits[i];
      analyzedCount++;
      
      try {
        // 1. 获取该 commit 时的文件内容
        const contentAtCommit = await getFileContentAtCommit(filePath, commit.hash);
        
        // 2. 根据文件类型进行分析
        let isMatch = false;
        
        if (context.type === 'solarwire-file') {
          // .solarwire 文件：直接解析并对比
          isMatch = analyzeSolarWireFile(contentAtCommit, context.title);
        } else {
          // .md 文件：解析 Markdown，提取代码块并对比
          isMatch = analyzeMarkdownSnippet(
            contentAtCommit, 
            context.snippetIndex!, 
            context.title
          );
        }
        
        if (isMatch) {
          matchingCommits.push(commit);
        }
        
        // 3. 每 5 个提交更新一次进度
        if (analyzedCount % 5 === 0) {
          postMessage({
            type: 'progress',
            data: {
              total: commits.length,
              processed: analyzedCount,
              currentCommit: commit.hash,
              matchingCommits: [...matchingCommits],
              status: 'running'
            }
          } as WorkerResponse);
        }
        
      } catch (error) {
        console.error('Failed to analyze commit', commit.hash, error);
        // 继续分析下一个
      }
    }
    
    // 4. 分析完成
    postMessage({
      type: 'complete',
      data: {
        matchingCommits,
        analyzedCount,
        total: commits.length
      }
    } as WorkerResponse);
    
  } catch (error) {
    postMessage({
      type: 'error',
      data: {
        message: '分析过程中发生错误',
        suggestion: '请检查 Git 仓库是否正常'
      }
    } as WorkerResponse);
  }
};

/**
 * 分析 .solarwire 文件
 */
function analyzeSolarWireFile(content: string, title: string): boolean {
  // 检查文件是否包含该 title
  if (!content.includes(title)) {
    return false;
  }
  
  // 解析 AST（简化版，实际应该使用 SolarWire 解析器）
  const elements = parseSolarWireElements(content);
  
  // 检查是否有元素的 !title 匹配
  return elements.some(el => el.title === title);
}

/**
 * 分析 Markdown 文件中的 SolarWire 代码块
 */
function analyzeMarkdownSnippet(
  content: string, 
  snippetIndex: number, 
  title: string
): boolean {
  // 1. 使用 marked 解析 Markdown
  const tokens = marked.lexer(content);
  
  // 2. 提取所有 solarwire 代码块
  const solarwireBlocks: { code: string; index: number }[] = [];
  let currentIndex = 0;
  
  for (const token of tokens) {
    if (token.type === 'code' && token.lang === 'solarwire') {
      solarwireBlocks.push({
        code: token.text,
        index: currentIndex++
      });
    }
  }
  
  // 3. 找到目标代码块
  const targetBlock = solarwireBlocks.find(
    block => block.index === snippetIndex
  );
  
  if (!targetBlock) {
    return false;
  }
  
  // 4. 检查代码块是否包含该 title
  if (!targetBlock.code.includes(title)) {
    return false;
  }
  
  // 5. 解析代码块 AST 并对比
  const elements = parseSolarWireElements(targetBlock.code);
  return elements.some(el => el.title === title);
}

/**
 * 简化版 SolarWire 元素解析
 * 实际应该使用 @solarwire/parser
 */
function parseSolarWireElements(content: string): Array<{ title?: string }> {
  const elements: Array<{ title?: string }> = [];
  
  // 简单的正则解析（实际应该使用正式解析器）
  const titleMatch = content.match(/^!title=(.+)/m);
  if (titleMatch) {
    elements.push({ title: titleMatch[1].trim() });
  }
  
  return elements;
}

/**
 * 辅助函数：从主线程获取文件内容
 */
async function getFileContentAtCommit(filePath: string, commitHash: string): Promise<string> {
  // 这个函数需要通过 postMessage 与主线程通信
  return new Promise((resolve, reject) => {
    const messageId = Date.now();
    
    // 发送请求到主线程
    postMessage({
      type: 'getFileContent',
      id: messageId,
      filePath,
      commitHash
    });
    
    // 监听响应
    const handler = (event: MessageEvent) => {
      if (event.data.type === 'getFileContentResponse' && event.data.id === messageId) {
        self.removeEventListener('message', handler);
        if (event.data.success) {
          resolve(event.data.content);
        } else {
          reject(new Error(event.data.error));
        }
      }
    };
    
    self.addEventListener('message', handler);
    
    // 超时处理
    setTimeout(() => {
      self.removeEventListener('message', handler);
      reject(new Error('Timeout getting file content'));
    }, 5000);
  });
}
```

### 4.4 VersionCache 缓存系统

**文件位置**: `editor/src/shared/cache/versionCache.ts`

**实现代码**:
```typescript
interface VersionCacheItem {
  commits: GitCommit[];
  timestamp: number;
  analyzedCount: number;
  context: {
    filePath: string;
    snippet?: { code: string; snippetIndex: number };
    title: string;
  };
}

class VersionCache {
  private cache = new Map<string, VersionCacheItem>();
  private readonly MAX_SIZE = 100; // 最多缓存 100 个文件
  private readonly TTL = 1000 * 60 * 30; // 30 分钟过期

  /**
   * 构建缓存 Key
   */
  buildCacheKey(
    filePath: string,
    snippet: { code: string; snippetIndex: number } | undefined,
    title: string
  ): string {
    if (snippet) {
      return `${filePath}:${snippet.snippetIndex}:${title}`;
    }
    return `${filePath}:${title}`;
  }

  /**
   * 获取缓存
   */
  get(key: string): VersionCacheItem | null {
    const item = this.cache.get(key);
    if (!item) return null;
    
    // 检查是否过期
    if (Date.now() - item.timestamp > this.TTL) {
      this.cache.delete(key);
      return null;
    }
    
    return item;
  }

  /**
   * 设置缓存
   */
  set(key: string, value: VersionCacheItem): void {
    // 如果缓存已满，删除最旧的
    if (this.cache.size >= this.MAX_SIZE) {
      const oldestKey = this.cache.keys().next().value;
      if (oldestKey) {
        this.cache.delete(oldestKey);
      }
    }
    
    this.cache.set(key, value);
  }

  /**
   * 使缓存失效（文件变化时调用）
   */
  invalidate(filePath: string): void {
    for (const key of this.cache.keys()) {
      if (key.startsWith(filePath)) {
        this.cache.delete(key);
      }
    }
  }

  /**
   * 清空缓存
   */
  clear(): void {
    this.cache.clear();
  }

  /**
   * 获取缓存统计信息
   */
  getStats(): { size: number; keys: string[] } {
    return {
      size: this.cache.size,
      keys: Array.from(this.cache.keys())
    };
  }
}

// 单例模式
const versionCache = new VersionCache();
export { versionCache };
```

```typescript
/**
 * 验证文件或代码块是否包含 !title

function validateTitle(
  filePath: string,
  snippet?: { code: string; snippetIndex: number }
): TitleValidationResult {
  try {
    if (snippet) {
      // Markdown 文件中的代码块
      const titleMatch = snippet.code.match(/^!title=(.+)/m);
      if (!titleMatch) {
        return {
          valid: false,
          error: '该代码块缺少 !title 标识',
          suggestion: `请在代码块第一行添加 !title=xxx
例如：
\`\`\`solarwire
!title=登录页面流程图
...\`\`\``
        };
      }
      return {
        valid: true,
        title: titleMatch[1].trim()
      };
    } else {
      // .solarwire 文件
      const content = await readFile(filePath);
      const titleMatch = content.match(/^!title=(.+)/m);
      if (!titleMatch) {
        return {
          valid: false,
          error: '该文件缺少 !title 标识',
          suggestion: `请在文件第一行添加 !title=xxx
例如：
!title=登录页面流程图
...`
        };
      }
      return {
        valid: true,
        title: titleMatch[1].trim()
      };
    }
  } catch (error) {
    return {
      valid: false,
      error: '无法读取文件内容',
      suggestion: '请确保文件存在且可访问'
    };
  }
}
```

### 6.2 AST 对比算法

```typescript
/**
 * 对比两个 AST 是否有变化
 */
function compareAST(
  oldAST: SolarWireAST,
  newAST: SolarWireAST,
  title: string
): boolean {
  // 1. 找到包含该 title 的元素
  const oldElements = oldAST.elements.filter(el => el.title === title);
  const newElements = newAST.elements.filter(el => el.title === title);
  
  // 2. 数量变化说明有修改
  if (oldElements.length !== newElements.length) {
    return true;
  }
  
  // 3. 对比每个元素的属性
  for (let i = 0; i < oldElements.length; i++) {
    if (!elementsEqual(oldElements[i], newElements[i])) {
      return true;
    }
  }
  
  return false;
}

/**
 * 对比两个元素是否相等
 */
function elementsEqual(el1: SolarWireElement, el2: SolarWireElement): boolean {
  // 对比类型
  if (el1.type !== el2.type) return false;
  
  // 对比位置
  if (el1.location?.line !== el2.location?.line) return false;
  
  // 对比属性
  const props1 = JSON.stringify(el1.properties);
  const props2 = JSON.stringify(el2.properties);
  if (props1 !== props2) return false;
  
  return true;
}
```

```typescript
/**
 * 从 Markdown 内容中提取第 N 个 SolarWire 代码块
 */
function extractNthSolarWireBlock(
  markdownContent: string,
  snippetIndex: number
): string | null {
  // 1. 使用 marked 解析 Markdown
  const tokens = marked.lexer(markdownContent);
  
  // 2. 提取所有 solarwire 代码块
  const solarwireBlocks: string[] = [];
  
  for (const token of tokens) {
    if (token.type === 'code' && token.lang === 'solarwire') {
      solarwireBlocks.push(token.text);
    }
  }
  
  // 3. 返回第 N 个代码块
  return solarwireBlocks[snippetIndex] || null;
}
```

### 8.3 Worker 优化

```typescript
// Worker 池（避免频繁创建/销毁）
class WorkerPool {
  private pool: Worker[] = [];
  private readonly MAX_SIZE = 3;

  acquire(): Worker {
    if (this.pool.length > 0) {
      return this.pool.pop()!;
    }
    return new Worker(
      new URL('../workers/git-diff-analyzer.worker.ts', import.meta.url)
    );
  }

  release(worker: Worker): void {
    if (this.pool.length < this.MAX_SIZE) {
      this.pool.push(worker);
    } else {
      worker.terminate();
    }
  }
}

const workerPool = new WorkerPool();
```
