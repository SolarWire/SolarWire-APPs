# 相对坐标支持方案设计

## 1. 背景与目标

### 1.1 背景

SolarWire 编辑器需要在**不修改渲染器核心**的前提下，在编辑器层面实现对相对坐标的完整支持。渲染器已经充分测试，支持以下坐标类型：

- `absolute`: 绝对坐标 `{ type: 'absolute', value: number }`
- `relative`: 相对坐标 `{ type: 'relative', value: number }`
- `edge`: 边缘参考坐标 `{ type: 'edge', direction: 'L'|'R'|'T'|'B'|'C', value: number }`
- 线段终点支持相对偏移：`{ dx: number, dy: number }`

### 1.2 目标

在编辑器层面实现：
1. **直观的相对坐标编辑**：用户可以在属性面板中切换绝对/相对坐标模式
2. **智能的拖动体验**：拖动元素时保持相对坐标语义不丢失
3. **线段特殊支持**：线段的起点和终点可以独立使用不同的坐标模式
4. **零渲染器改动**：所有转换在编辑器层完成，渲染器保持现有逻辑

### 1.3 设计决策确认

- **新建元素默认行为**：默认使用绝对坐标（相对于画布原点）
- **拖动参考点**：相对于画布原点 (0, 0)
- **属性面板模式**：完整模式，始终显示坐标类型和参考元素选择

## 3. 实现细节

### 3.1 坐标转换工具函数

**文件**: `src/shared/utils/coordinate-converter.ts`

```typescript
import { Coordinate, Element, ElementBounds } from '../../lib/parser/types';

/**
 * 将绝对坐标转换为相对坐标（相对于参考点）
 * @param absolute 绝对坐标 { x, y }
 * @param reference 参考点坐标 { x, y }
 * @returns 相对偏移 { dx, dy }
 */
export function absoluteToRelative(
  absolute: { x: number; y: number },
  reference: { x: number; y: number }
): { dx: number; dy: number } {
  return {
    dx: absolute.x - reference.x,
    dy: absolute.y - reference.y
  };
}

/**
 * 将相对坐标转换为绝对坐标
 * @param relative 相对偏移 { dx, dy }
 * @param reference 参考点坐标 { x, y }
 * @returns 绝对坐标 { x, y }
 */
export function relativeToAbsolute(
  relative: { dx: number; dy: number },
  reference: { x: number; y: number }
): { x: number; y: number } {
  return {
    x: reference.x + relative.dx,
    y: reference.y + relative.dy
  };
}

/**
 * 标准化坐标：将任意坐标类型转换为绝对值
 * @param coord 坐标对象（absolute/relative/edge）
 * @param referenceBounds 参考元素边界（用于 edge 坐标）
 * @param defaultValue 默认值
 * @returns 绝对值
 */
export function normalizeCoordinate(
  coord: Coordinate,
  referenceBounds: ElementBounds | null,
  defaultValue: number = 0
): number {
  switch (coord.type) {
    case 'absolute':
      return coord.value;
    
    case 'relative':
      // 相对坐标需要参考点，这里返回相对值本身
      // 实际使用时需要加上参考点的坐标
      return coord.value;
    
    case 'edge':
      if (!referenceBounds) {
        return defaultValue;
      }
      switch (coord.direction) {
        case 'L': return referenceBounds.x + coord.value;
        case 'R': return referenceBounds.x + referenceBounds.width + coord.value;
        case 'T': return referenceBounds.y + coord.value;
        case 'B': return referenceBounds.y + referenceBounds.height + coord.value;
        case 'C': return referenceBounds.x + referenceBounds.width / 2 + coord.value;
        default: return defaultValue;
      }
    
    default:
      const exhaustiveCheck: never = coord;
      throw new Error(`Unknown coordinate type: ${(exhaustiveCheck as any).type}`);
  }
}

/**
 * 获取元素的边界信息
 * @param element 元素对象
 * @param lastElementBounds 上一个元素的边界（用于相对坐标计算）
 * @returns 元素边界
 */
export function getElementBounds(
  element: Element,
  lastElementBounds: ElementBounds | null = null
): ElementBounds {
  const attrs = element.attributes;
  
  // 处理坐标
  let x = 0, y = 0;
  if (element.coordinates) {
    x = normalizeCoordinate(element.coordinates.x, lastElementBounds, 0);
    y = normalizeCoordinate(element.coordinates.y, lastElementBounds, 0);
  }
  
  // 处理尺寸
  const w = parseInt(attrs.w) || 100;
  const h = parseInt(attrs.h) || 100;
  
  return { x, y, width: w, height: h };
}

/**
 * 将元素坐标更新为新的绝对坐标
 * @param element 元素对象
 * @param newX 新的 X 坐标
 * @param newY 新的 Y 坐标
 * @returns 更新后的元素对象（浅拷贝）
 */
export function updateElementCoordinates(
  element: Element,
  newX: number,
  newY: number
): Element {
  return {
    ...element,
    coordinates: {
      x: { type: 'absolute', value: newX },
      y: { type: 'absolute', value: newY }
    }
  };
}

/**
 * 将线段终点更新为相对偏移
 * @param lineElement 线段元素
 * @param dx X 方向偏移
 * @param dy Y 方向偏移
 * @returns 更新后的线段元素
 */
export function updateLineEndRelative(
  lineElement: Element,
  dx: number,
  dy: number
): Element {
  if (lineElement.type !== 'line') {
    return lineElement;
  }
  
  const lineEl = lineElement as any;
  return {
    ...lineEl,
    end: { type: 'relative', dx, dy }
  };
}

/**
 * 将线段终点更新为绝对坐标
 * @param lineElement 线段元素
 * @param x 终点 X 坐标
 * @param y 终点 Y 坐标
 * @returns 更新后的线段元素
 */
export function updateLineEndAbsolute(
  lineElement: Element,
  x: number,
  y: number
): Element {
  if (lineElement.type !== 'line') {
    return lineElement;
  }
  
  const lineEl = lineElement as any;
  return {
    ...lineEl,
    end: {
      x: { type: 'absolute', value: x },
      y: { type: 'absolute', value: y }
    }
  };
}
```

### 3.2 拖动坐标处理 Hook

**文件**: `src/shared/hooks/useDragCoordinate.ts`

```typescript
import { useState, useCallback, useRef } from 'react';
import { ElementBounds } from '../../lib/parser/types';
import { absoluteToRelative, relativeToAbsolute } from '../utils/coordinate-converter';

interface UseDragCoordinateOptions {
  initialMode: 'absolute' | 'relative';
  referenceBounds?: ElementBounds;
}

interface DragCoordinateState {
  mode: 'absolute' | 'relative';
  initialAbsolute: { x: number; y: number };
  initialRelative?: { dx: number; dy: number };
  referenceBounds?: ElementBounds;
}

interface UseDragCoordinateReturn {
  state: DragCoordinateState | null;
  startDrag: (
    x: number,
    y: number,
    referenceBounds?: ElementBounds
  ) => void;
  updateDrag: (newX: number, newY: number) => {
    x: number;
    y: number;
    isRelative: boolean;
  };
  endDrag: () => void;
  setMode: (mode: 'absolute' | 'relative') => void;
}

export function useDragCoordinate(
  options: UseDragCoordinateOptions
): UseDragCoordinateReturn {
  const [state, setState] = useState<DragCoordinateState | null>(null);
  const modeRef = useRef<'absolute' | 'relative'>(options.initialMode);

  const startDrag = useCallback((
    x: number,
    y: number,
    referenceBounds?: ElementBounds
  ) => {
    const initialRelative = referenceBounds
      ? absoluteToRelative({ x, y }, referenceBounds)
      : undefined;

    setState({
      mode: modeRef.current,
      initialAbsolute: { x, y },
      initialRelative,
      referenceBounds
    });
  }, []);

  const updateDrag = useCallback((newX: number, newY: number) => {
    if (!state) {
      return { x: newX, y: newY, isRelative: false };
    }

    if (state.mode === 'absolute') {
      return { x: newX, y: newY, isRelative: false };
    } else {
      // 相对模式：计算相对于参考点的偏移
      if (state.referenceBounds && state.initialRelative !== undefined) {
        const newRelative = absoluteToRelative({ x: newX, y: newY }, state.referenceBounds);
        return {
          x: newRelative.dx,
          y: newRelative.dy,
          isRelative: true
        };
      }
      return { x: newX, y: newY, isRelative: false };
    }
  }, [state]);

  const endDrag = useCallback(() => {
    setState(null);
  }, []);

  const setMode = useCallback((mode: 'absolute' | 'relative') => {
    modeRef.current = mode;
    if (state) {
      setState({
        ...state,
        mode
      });
    }
  }, [state]);

  return {
    state,
    startDrag,
    updateDrag,
    endDrag,
    setMode
  };
}
```

### 3.5 线段元素特殊处理

线段元素有起点和终点两个坐标，需要分别处理：

```typescript
interface LineCoordinateState {
  startMode: 'absolute' | 'relative';
  endMode: 'absolute' | 'relative';
  referenceBounds?: ElementBounds;
}

// 在 SolarWireView 中添加线段拖动处理
const handleLineDrag = useCallback((
  elementId: string,
  handle: 'start' | 'end',
  newWorldX: number,
  newWorldY: number
) => {
  const elementData = getElementData(elementId);
  if (!elementData || elementData.type !== 'line') return;
  
  const lineElement = elementData as any;
  
  // 获取起点坐标
  let startX = 0, startY = 0;
  if (lineElement.start.x.type === 'absolute') {
    startX = lineElement.start.x.value;
  }
  if (lineElement.start.y.type === 'absolute') {
    startY = lineElement.start.y.value;
  }
  
  if (handle === 'start') {
    // 拖动起点
    if (lineElement.end && 'dx' in lineElement.end && 'dy' in lineElement.end) {
      // 终点是相对坐标：需要同时更新起点和保持相对偏移
      const dx = lineElement.end.dx;
      const dy = lineElement.end.dy;
      
      updateLineElement(elementId, {
        start: {
          x: { type: 'absolute', value: newWorldX },
          y: { type: 'absolute', value: newWorldY }
        },
        end: { type: 'relative', dx, dy } // 保持相对偏移
      });
    } else {
      // 终点是绝对坐标：只更新起点
      updateLineElement(elementId, {
        start: {
          x: { type: 'absolute', value: newWorldX },
          y: { type: 'absolute', value: newWorldY }
        }
      });
    }
  } else if (handle === 'end') {
    // 拖动终点
    if (lineElement.end && 'dx' in lineElement.end && 'dy' in lineElement.end) {
      // 终点是相对坐标：计算新的相对偏移
      const newDx = newWorldX - startX;
      const newDy = newWorldY - startY;
      
      updateLineElement(elementId, {
        end: { type: 'relative', dx: newDx, dy: newDy }
      });
    } else {
      // 终点是绝对坐标：直接更新
      updateLineElement(elementId, {
        end: {
          x: { type: 'absolute', value: newWorldX },
          y: { type: 'absolute', value: newWorldY }
        }
      });
    }
  }
}, []);
```

## 4. 线段元素特殊处理

### 4.1 线段坐标模式组合

线段元素的起点和终点可以独立使用不同的坐标模式：

| 起点模式 | 终点模式 | 使用场景 |
|---------|---------|---------|
| 绝对 | 绝对 | 精确定位的线段 |
| 绝对 | 相对 | 从固定点出发，长度可变的线段 |
| 相对 | 绝对 | 连接到固定点的线段 |
| 相对 | 相对 | 保持相对位置和长度的线段 |

### 4.2 拖动行为

**拖动起点**：
- 如果终点是相对坐标：保持相对偏移不变，只更新起点
- 如果终点是绝对坐标：只更新起点，终点不变

**拖动终点**：
- 如果终点是相对坐标：更新相对偏移量
- 如果终点是绝对坐标：直接更新终点坐标

**拖动线段中间**：
- 整体移动线段，起点和终点同时更新
- 如果终点是相对坐标：保持相对偏移不变

## 11. 结论

本方案通过在编辑器层面添加坐标转换中间件，实现了对相对坐标的完整支持，同时保持渲染器核心不变。方案具有以下特点：

- **渐进式实现**：可以逐步添加功能，不影响现有代码
- **向后兼容**：现有绝对坐标功能不受影响
- **线段特殊支持**：起点和终点可以独立使用不同的坐标模式
- **可扩展性**：易于添加新的坐标类型和参考模式

通过实现本方案，SolarWire 编辑器将提供更强大的坐标编辑能力，同时保持渲染器的稳定性和可靠性。
